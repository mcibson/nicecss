Basic Example

@mixin container();

  • @include container__padding;
  • display:flex;
  • flex-wrap: wrap;
  • flex-direction: row;
  • width: 100%;
  • max-width: $maxWidth;
  • margin: 0 auto $container__margin-bottom;

Boxed Container

enable $outer to add outer paddings to the container

@mixin grid();

  • @include padding;
  • @include width($default, $s, $m, $l, $xl, $xxl, $cols );
  • @include jacket(ie) { float: left; }

Boxed Container

enable $outer to add outer paddings to the container

// Code Example:

.posts {
    @include container();
}
.post {
    @include grid(12, $m:6; $s:4);
}
// Output:

.posts {
    display: flex;
    flex-wrap: wrap;
    flex-direction: row;
    width: 100%;
    max-width: 1200px;
    margin: 0 auto 1rem;
}

.post {
    padding: 1rem;
    width: 100%;
    @media (min-width: 480) {
        width: 50%;
    }
    @media (min-width: 768px) {
        width: 33.33333%;
    }
}

Basic Card Example

@mixin card();

  • @include margin()
  • @include padding;
  • @include width--card($default, $s, $m, $l, $xl, $xxl, $cols);
  • @include jacket(ie) { float: left; }

Basic Card width paddings and margins around

IE9+ - uses calc() to get the width

@mixin card--default();

  • @include margin--default()
  • @include padding;
  • @include width--card($default, $s, $m, $l, $xl, $xxl, $cols);
  • @include jacket(ie) { float: left; }

margin--default

has margin on sides and bottom. Margin bottom is doubled to get the equal vertical and horizontal whitespace

// Code Example:

.posts {
    @include container();
}
.post--card {
    @include grid(12, $m:6; $s:4);
}
// Output:

.posts {
    display: flex;
    flex-wrap: wrap;
    flex-direction: row;
    width: 100%;
    max-width: 1200px;
    margin: 0 auto 1rem;
}

.post--card {
    margin: 1rem;
    padding: 1rem;
    width: calc(100% - 2rem);
    @media (min-width: 480px) {
        width: calc(50% - 2rem);
    }
    @media (min-width: 768px) {
        width: calc(33.33333% - 2rem);
    }
}
